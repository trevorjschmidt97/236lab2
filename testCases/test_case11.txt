    nums(A, B)

    add('1', '2', '3').
    add('2','4','5').
    nums('2', '3').
    nums('6', '7').

Rules:
    add(X,Y,Z) :- nums(X,Y), add(X,Y,((X+Y)Z)).

    add(X,Y,'13')?
